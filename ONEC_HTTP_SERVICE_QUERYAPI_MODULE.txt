Функция QueryPOST(Запрос) Экспорт
	Попытка
		Тело = Запрос.ПолучитьТелоКакСтроку();
		Чтение = Новый ЧтениеJSON;
		Чтение.УстановитьСтроку(Тело);
		Данные = ПрочитатьJSON(Чтение);

		Если ТипЗнч(Данные) <> Тип("Структура") Или Не Данные.Свойство("query") Тогда
			Возврат ОтветJSON(400, Новый Структура("error", "query required"));
		КонецЕсли;

		ТекстЗапроса = СокрЛП(Строка(Данные.query));
		Если ПустаяСтрока(ТекстЗапроса) Тогда
			Возврат ОтветJSON(400, Новый Структура("error", "query is empty"));
		КонецЕсли;

		Запрос1С = Новый Запрос;
		Запрос1С.Текст = ТекстЗапроса;
		Таблица = Запрос1С.Выполнить().Выгрузить();

		Rows = Новый Массив;
		Для Каждого Стр Из Таблица Цикл
			Obj = Новый Структура;
			Для Каждого Кол Из Таблица.Колонки Цикл
				Знач = Стр[Кол.Имя];
				Obj.Вставить(Кол.Имя, ЗначениеДляJSON(Знач));
			КонецЦикла;
			Rows.Добавить(Obj);
		КонецЦикла;

		Возврат ОтветJSON(200, Новый Структура("rows", Rows));
	Исключение
		Возврат ОтветJSON(500, Новый Структура("error", ОписаниеОшибки()));
	КонецПопытки;
КонецФункции

Функция ОтветJSON(Код, Данные) Экспорт
	Запись = Новый ЗаписьJSON;
	Запись.УстановитьСтроку();
	ЗаписатьJSON(Запись, Данные);
	Текст = Запись.Закрыть();

	Ответ = Новый HTTPСервисОтвет(Код);
	Ответ.Заголовки.Вставить("Content-Type", "application/json; charset=utf-8");
	Ответ.УстановитьТелоИзСтроки(Текст);
	Возврат Ответ;

Функция ПреобразоватьСхемуВТекст(Объекты) Экспорт
	Строки = Новый Массив;

	Для Каждого Объект Из Объекты Цикл
		Поля = "";
		Если Объект.fields.Количество() > 0 Тогда
			Поля = СтрСоединить(Объект.fields, ", ");
		КонецЕсли;
		Строки.Добавить(Объект.type + "." + Объект.name + "(" + Поля + ")");
	КонецЦикла;

	Если Строки.Количество() = 0 Тогда
		Возврат "";
	КонецЕсли;

	Возврат СтрСоединить(Строки, ";" + Символы.ПС) + ";";

Процедура ДобавитьКоллекцию(Результат, ИмяТипа, Коллекция) Экспорт
	Для Каждого ОбъектМД Из Коллекция Цикл
		Поля = Новый Массив;

		Если ЕстьРеквизит(ОбъектМД, "Ссылка") Тогда
			ДобавитьПолеЕслиНет(Поля, "Ссылка");
		КонецЕсли;

		// Читаем коллекции безопасно: у разных типов метаданных свойства различаются.
		ДобавитьПоляПоИмениСвойства(Поля, ОбъектМД, "Измерения");
		ДобавитьПоляПоИмениСвойства(Поля, ОбъектМД, "Ресурсы");
		ДобавитьПоляПоИмениСвойства(Поля, ОбъектМД, "Реквизиты");
		// Табличные части не добавляем как обычные поля, чтобы не порождать невалидные SELECT-поля.

		Если ЕстьРеквизит(ОбъектМД, "Наименование") Тогда
			ДобавитьПолеЕслиНет(Поля, "Наименование");
		КонецЕсли;
		Если ЕстьРеквизит(ОбъектМД, "Код") Тогда
			ДобавитьПолеЕслиНет(Поля, "Код");
		КонецЕсли;
		Если ЕстьРеквизит(ОбъектМД, "Дата") Тогда
			ДобавитьПолеЕслиНет(Поля, "Дата");
		КонецЕсли;
		Если ЕстьРеквизит(ОбъектМД, "Номер") Тогда
			ДобавитьПолеЕслиНет(Поля, "Номер");
		КонецЕсли;

		Результат.Добавить(Новый Структура("type,name,fields", ИмяТипа, ОбъектМД.Имя, Поля));
	КонецЦикла;
КонецПроцедуры

Процедура ДобавитьПоляПоИмениСвойства(Поля, ОбъектМД, ИмяСвойства) Экспорт
	Попытка
		Коллекция = Неопределено;
		Если ИмяСвойства = "Измерения" Тогда
			Коллекция = ОбъектМД.Измерения;
		ИначеЕсли ИмяСвойства = "Ресурсы" Тогда
			Коллекция = ОбъектМД.Ресурсы;
		ИначеЕсли ИмяСвойства = "Реквизиты" Тогда
			Коллекция = ОбъектМД.Реквизиты;
		ИначеЕсли ИмяСвойства = "ТабличныеЧасти" Тогда
			Коллекция = ОбъектМД.ТабличныеЧасти;
		Иначе
			Возврат;
		КонецЕсли;

		Для Каждого Элемент Из Коллекция Цикл
			ДобавитьПолеЕслиНет(Поля, Элемент.Имя);
		КонецЦикла;
	Исключение
		// У данного типа объекта нет такого свойства.
	КонецПопытки;
КонецПроцедуры

Функция ЗначениеДляJSON(Знач) Экспорт
	Т = ТипЗнч(Знач);
	Если Т = Тип("Строка")
		Или Т = Тип("Число")
		Или Т = Тип("Булево")
		Или Т = Тип("Неопределено") Тогда
		Возврат Знач;
	КонецЕсли;

	Если Т = Тип("Дата") Тогда
		Возврат Формат(Знач, "ДФ=yyyy-MM-dd HH:mm:ss");
	КонецЕсли;

	Возврат Строка(Знач);
